import java.io.*;
import java.math.*;
import java.security.*;
import java.text.*;
import java.util.*;
import java.util.concurrent.*;
import java.util.function.*;
import java.util.regex.*;
import java.util.stream.*;
import static java.util.stream.Collectors.joining;
import static java.util.stream.Collectors.toList;

class Result {

    public static void insertionSort1(int n, List<Integer> arr) {
        int valueToInsert = arr.get(n - 1);
        int i = n - 2;

        // Keep moving back until we find where to place the valueToInsert
        while (i >= 0 && arr.get(i) > valueToInsert) {
            arr.set(i + 1, arr.get(i));
            printArray(arr); // Printing the list after every shift
            i--;
        }

        arr.set(i + 1, valueToInsert);
        printArray(arr); // Printing the list after inserting the value
    }

    // Utility function to print the array
    private static void printArray(List<Integer> arr) {
        for (int num : arr) {
            System.out.print(num + " ");
        }
        System.out.println();
    }
}
public class Solution {
    public static void main(String[] args) throws IOException {
        BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(bufferedReader.readLine().trim());

        List<Integer> arr = Stream.of(bufferedReader.readLine().replaceAll("\\s+$", "").split(" "))
            .map(Integer::parseInt)
            .collect(toList());

        Result.insertionSort1(n, arr);

        bufferedReader.close();
    }
}
